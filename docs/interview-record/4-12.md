# 三道简单js题

## 1、闭包概念

* 给定一个数和一个方法vote，实现一个方法before，使得多次调用vote时小于给定的数

```js
// 用例
const before = (n,fn)=>{
    // 代码
}
  const vote = before(3, function (name) {
        console.log(`${name}执行`)
      })

      vote('A') // A执行
      vote('B') // B执行
      vote('C') // C执行
      vote('D')
      vote('E')
```
* 题解
```js
const before = (n,fn) => {
    return (m)=>{
        if(n>0){
            fn(m)
        }
        n--
    }
}
```
## 2、promiseAll实现

* 写一个方法promiseAll，实现类似有promise.all()的效果

```js
// 用例

const promiseAll = (arr)=>{
    // 代码
}
let promise1 = Promise.resole('1')
let promise2 = Promise.resole('2')
let promise3 = Promise.resole('3') 
promiseAll([promise1,promise2,promise3]).then((value)=>{
    console.log(value)
})
```
* 题解
```js
const promiseAll = (arr) => {
        if (!Array.isArray(arr)) {
          throw new TypeError('promises mast be array')
        }
        if (arr.length === 0) {
          return Promise.resolve([])
        }
        let count = 0
        const resultList = []
        return new Promise((resolve, reject) => {
          for (let i = 0; i < arr.length; i++) {
            arr[i]
              .then((value) => {
                resultList.push(value)
                count++
                if (count === arr.length) resolve(resultList)
              })
              .catch((error) => {
                reject(error)
              })
          }
        })
      }
```
## 3、数组方法

* 给定两个整数数组A，B，写一个方法findExistEleIndex()，找出A中等于B中元素下标

```js
// 用例

const findExistEleIndex = (A，B)=>{
    // 代码
}
const A = [1,2,4,2,4]
const B = [1,4]
 // 共同的元素是1和4, 在A中的下标是 [0, 2, 4]
```
* 题解
```js
    // 简单思路
    const findExistEleIndex = (A,B) =>{
        if(B.length === 0 || A.length === 0){
            return []
        }
        let result = []
        for(let i = 0; i<A.length; i++) {
            if(B.includes(A[i])){
                result.push(i)
            }
        }
        return result
    }
    // 使用set优化
    const findExistEleIndex = (A,B) =>{
        if(B.length === 0 || A.length === 0){
            return []
        }
        const setB = new Set(B)
        let result = []
        for(let i = 0; i<A.length; i++) {
            if(setB.has(A[i])){
                result.push(i)
            }
        }
        return result
    }
```
